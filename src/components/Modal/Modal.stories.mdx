 <Story name="- Basic">
    {() => {
      const [modalOpen, setModalOpen] = useState(false);
      return (
        <>
          <Button onClick={() => setModalOpen(true)}>
            Open Basic Dialog
          </Button>
          <StackModal
            title="Delete service from application"
            isOpen={modalOpen}
            onClose={() => setModalOpen(false)}>
             
              <p>Are you sure you want to remove this service from the application?</p>
             
             
              <Button variant="secondary" onClick={() => setModalOpen(false)}>
                Cancel
              </Button>
              <Button onClick={() => setModalOpen(false)}>
                Delete
              </Button>
             
          </StackModal>
        </>
      );
    }}
  </Story>
