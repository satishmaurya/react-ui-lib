import { useState, useEffect } from 'react';
import { Meta, Story, Preview, ColorPalette, ColorItem, } from '@storybook/addon-docs/blocks';
import styled from 'styled-components';

import { IconNames } from "@blueprintjs/icons";

import Icon from '../../src/components/Icon/Icon';
import Button from '../../src/components/Button/Button'
import SearchBox from '../../src/components/SearchBox/SearchBox';

export const IconGallery = () => {
  const icons = Object.keys(IconNames).map(name => ({ iconName: IconNames[name]}));
  const [searchTerm, setSearchTerm] = React.useState("");
  const [searchResults, setSearchResults] = React.useState([]);
  const handleChange = event => {
    console.log(event.target.value, 'val');
   setSearchTerm(event.target.value);
  };
   const copyCodeToClipboard = () => {
    const el = this.iconName
    el.select()
    document.execCommand("copy")
  }
  React.useEffect(() => {
    const results = icons.filter(icon =>
      icon.iconName.toLowerCase().includes(searchTerm.toLowerCase())
    );
    setSearchResults(results);
  }, [searchTerm]);
  return (
    <>
       <SearchBox type="text" placeholder="Search" defaultValue={searchTerm} onChange={handleChange} style={{width:'100%'}} />
      <IconContainer>
         {searchResults.map(item => (
          <IconItem key={item.iconName} data-name={item.iconName}><Icon size={16} icon={item.iconName} onClick={copyCodeToClipboard}/></IconItem>
        ))}
      </IconContainer>
    </>
  );
}

export const IconContainer = styled.div`
    display: flex;
    width: 100%;
    flex-wrap: wrap;
    margin-bottom: 20px;
    margin-top: 20px;
    box-shadow: inset 0 0 0 -1px #000;
`;

    background: #fff;

    }
    &::after {
        content: attr(data-name);
        opacity: 0;
        position: absolute;
        font-size: 12px;
        top: -15px;
        white-space: nowrap;
        left: 50%;
        color: #000;
        padding: 4px 8px;
        background: #ddd;

        transform: translateX(-50%);
    }
    &:hover::before,
    &:hover::after {
        opacity: 1;
    }
`;

export const IconPalette = () => {
    return icons.map(icon => (
        <IconItem key={icon.iconName} data-name={icon.iconName}><Icon size={16} icon={icon.iconName} /></IconItem>
    ))
}
